Data Viz Art Musings#

Sometimes my viz produces unpredictable outputs, and might as well collect offshoots of those.

import altair as alt
import polars as pl

Create your dataframe#

Think I can work more robust version, but for the sake of simplicity

df = pl.DataFrame(
    {
        'arc': [1, 1, 1, 1, 1, 1, 1, 1],
        'class': ["3", "3", "3", "5", "5", "5", "5", "5"],
        'subclass': ["3A", "3B", "3C", "5A", "5B", "5C", "5D", "5E"]
    }
)

First Arc#

Manipulating these Radius settings is key

arc_1 = (
    alt.Chart(df).mark_arc(
        innerRadius=25,
        outerRadius=60,
        cornerRadius=70,
        strokeOpacity=1,
        stroke="white",
        strokeWidth=2,
        binSpacing=0
    )
    .encode(
        theta=alt.Theta(field='arc', type='quantitative'),
        color="class:N"
    )
)

Second Arc#

arc_2 = (
    alt.Chart(df).mark_arc(
        innerRadius=10,
        outerRadius=23,
        cornerRadius=10,
        strokeOpacity=1,
        stroke="white",
        strokeWidth=2,
        binSpacing=0
    )
    .encode(
        theta=alt.Theta(field='arc', type='quantitative'),
        color=alt.Color("subclass:N")
    )
)

arc_3 = (
    alt.Chart(df).mark_arc(
        innerRadius=88,
        outerRadius=100,
        cornerRadius=10,
        strokeOpacity=1,
        stroke="white",
        strokeWidth=2,
        binSpacing=0
    )
    .encode(
        theta=alt.Theta(field='arc', type='quantitative'),
        color=alt.Color("arc:O")
    )
)

test = ((arc_1 + arc_2 + arc_3)
    .configure_view(
        fill="gray",
        fillOpacity=0.1,
        step=0,
        strokeWidth=1,
        stroke="lightgray",
    )).properties(
        title=alt.Title(
            text="Three pie charts.",
            font="Georgia",
            fontSize=20
        )
    )

test

test.save("pie_chart.html")
